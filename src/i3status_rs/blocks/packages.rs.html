1//! Pending updates for different package manager like apt, pacman, etc.
2//!
3//! Currently, these package managers are supported:
4//! - `apk` for Alpine Linux
5//! - `apt` for Debian/Ubuntu-based systems
6//! - `aur` for Arch-based systems
7//! - `brew` for the Homebrew Package Manager
8//! - `dnf` for Fedora-based systems
9//! - `pacman` for Arch-based systems
10//! - `snap` for Snap packages
11//! - `xbps` for Void Linux
12//!
13//! # Configuration
14//!
15//! Key | Values | Default
16//! ----|--------|--------
17//! `interval` | Update interval in seconds. | `600`
18//! `package_manager` | Package manager to check for updates | Automatically derived from format templates, but can be used to influence the `$total` value
19//! `format` | A string to customise the output of this block. See below for available placeholders. | `" $icon $total.eng(w:1) "`
20//! `format_singular` | Same as `format`, but for when exactly one update is available. | `" $icon $total.eng(w:1) "`
21//! `format_up_to_date` | Same as `format`, but for when no updates are available. | `" $icon $total.eng(w:1) "`
22//! `warning_updates_regex` | Display block as warning if updates matching regex are available. | `None`
23//! `critical_updates_regex` | Display block as critical if updates matching regex are available. | `None`
24//! `ignore_updates_regex` | Doesn't include updates matching regex in the count. | `None`
25//! `ignore_phased_updates` | Doesn't include potentially held back phased updates in the count. (For Debian/Ubuntu-based systems) | `false`
26//! `aur_command` | AUR command to check available updates, which outputs in the same format as pacman. E.g. `yay -Qua` (For Arch-based systems) | Required if `$aur` is used
27//!
28//!  Placeholder | Value                                                                            | Type   | Unit
29//! -------------|----------------------------------------------------------------------------------|--------|-----
30//! `icon`       | A static icon                                                                    | Icon   | -
31//! `apk`        | Number of updates available in Alpine Linux                                      | Number | -
32//! `apt`        | Number of updates available in Debian/Ubuntu-based systems                       | Number | -
33//! `aur`        | Number of updates available in Arch-based systems                                | Number | -
34//! `brew`       | Number of updates available in the Homebrew Package Manager                      | Number | -
35//! `dnf`        | Number of updates available in Fedora-based systems                              | Number | -
36//! `pacman`     | Number of updates available in Arch-based systems                                | Number | -
37//! `snap`       | Number of updates available in Snap packages                                     | Number | -
38//! `xbps`       | Number of updates available in Void Linux                                        | Number | -
39//! `total`      | Number of updates available in all package manager listed                        | Number | -
40//!
41//! # Apt
42//!
43//! Behind the scenes this uses `apt`, and in order to run it without root privileges i3status-rust will create its own package database in `/tmp/i3rs-apt/` which may take up several MB or more. If you have a custom apt config then this block may not work as expected - in that case please open an issue.
44//!
45//! Tip: You can grab the list of available updates using `APT_CONFIG=/tmp/i3rs-apt/apt.conf apt list --upgradable`
46//!
47//! # Pacman
48//!
49//! Requires fakeroot to be installed (only required for pacman).
50//!
51//! Tip: You can grab the list of available updates using `fakeroot pacman -Qu --dbpath /tmp/checkup-db-i3statusrs-$USER/`.
52//! If you have the `CHECKUPDATES_DB` env var set on your system then substitute that dir instead.
53//!
54//! Note: `pikaur` may hang the whole block if there is no internet connectivity [reference](https://github.com/actionless/pikaur/issues/595). In that case, try a different AUR helper.
55//!
56//! ### Pacman hook
57//!
58//! Tip: On Arch Linux you can setup a `pacman` hook to signal i3status-rs to update after packages
59//! have been upgraded, so you won't have stale info in your pacman block.
60//!
61//! In the block configuration, set `signal = 1` (or another number if `1` is being used by some
62//! other block):
63//!
64//! ```toml
65//! [[block]]
66//! block = "packages"
67//! signal = 1
68//! ```
69//!
70//! Create `/etc/pacman.d/hooks/i3status-rust.hook` with the below contents:
71//!
72//! ```ini
73//! [Trigger]
74//! Operation = Upgrade
75//! Type = Package
76//! Target = *
77//!
78//! [Action]
79//! When = PostTransaction
80//! Exec = /usr/bin/pkill -SIGRTMIN+1 i3status-rs
81//! ```
82//!
83//! # Example
84//!
85//! Apk-only config:
86//!
87//! ```toml
88//! [[block]]
89//! block = "packages"
90//! package_manager = ["apk"]
91//! interval = 1800
92//! error_interval = 300
93//! max_retries = 5
94//! format = " $icon $apk.eng(w:1) updates available "
95//! format_singular = " $icon One update available "
96//! format_up_to_date = " $icon system up to date "
97//! [[block.click]]
98//! # shows dmenu with available updates. Any dmenu alternative should also work.
99//! button = "left"
100//! cmd = "apk --no-cache --upgradable list | dmenu -l 10"
101//! ```
102//!
103//! Apt-only config
104//!
105//! ```toml
106//! [[block]]
107//! block = "packages"
108//! interval = 1800
109//! error_interval = 300
110//! max_retries = 5
111//! package_manager = ["apt"]
112//! format = " $icon $apt updates available"
113//! format_singular = " $icon One update available "
114//! format_up_to_date = " $icon system up to date "
115//! [[block.click]]
116//! # shows dmenu with cached available updates. Any dmenu alternative should also work.
117//! button = "left"
118//! cmd = "APT_CONFIG=/tmp/i3rs-apt/apt.conf apt list --upgradable | tail -n +2 | rofi -dmenu"
119//! [[block.click]]
120//! # Updates the block on right click
121//! button = "right"
122//! update = true

160//!
161//! Pacman-only config:

165//! block = "packages"
166//! package_manager = ["pacman"]

170//! format = " $icon $pacman updates available "
171//! format_singular = " $icon $pacman update available "
172//! format_up_to_date = " $icon system up to date "
173//! [[block.click]]
174//! # pop-up a menu showing the available updates. Replace wofi with your favourite menu command.
175//! button = "left"
176//! cmd = "fakeroot pacman -Qu --dbpath /tmp/checkup-db-i3statusrs-$USER/ | wofi --show dmenu"
177//! [[block.click]]
178//! # Updates the block on right click
179//! button = "right"

186//! [[block]]
187//! block = "packages"

193//! format_singular = " $icon $total update available "
194//! format_up_to_date = " $icon system up to date "
195//! # aur_command should output available updates to stdout (ie behave as echo -ne "update\n")
196//! aur_command = "yay -Qua"
197//! ```

216//!
217//! Xbps-only config:
218//!

220//! [[block]]
221//! block = "packages"

223//! interval = 1800

230//! # shows dmenu with available updates. Any dmenu alternative should also work.
231//! button = "left"
232//! cmd = "xbps-install -Mun | dmenu -l 10"
233//! ```
234//!
235//! Multiple package managers config:
236//!
237//! Update the list of pending updates every thirty minutes (1800 seconds):
238//!
239//! ```toml
240//! [[block]]
241//! block = "packages"
242//! package_manager = ["apk", "apt", "aur", "brew", "dnf", "pacman", "snap", "xbps"]
243//! interval = 1800
244//! error_interval = 300
245//! max_retries = 5
246//! format = " $icon $apk + $apt + $aur + $brew + $dnf + $pacman + $snap + $xbps = $total updates available "
247//! format_singular = " $icon One update available "
248//! format_up_to_date = " $icon system up to date "
249//! # If a linux update is available, but no ZFS package, it won't be possible to
250//! # actually perform a system upgrade, so we show a warning.
251//! warning_updates_regex = "(linux|linux-lts|linux-zen)"
252//! # If ZFS is available, we know that we can and should do an upgrade, so we show
253//! # the status as critical.
254//! critical_updates_regex = "(zfs|zfs-lts)"
255//! ```
256//!

286#[derive(Deserialize, Debug, SmartDefault, Clone)]
287#[serde(deny_unknown_fields, default)]
288pub struct Config {
289    #[default(600.into())]
290    pub interval: Seconds,
291    pub package_manager: Vec<PackageManager>,
292    pub format: FormatConfig,
293    pub format_singular: FormatConfig,
294    pub format_up_to_date: FormatConfig,
295    pub warning_updates_regex: Option<String>,
296    pub critical_updates_regex: Option<String>,
297    pub ignore_updates_regex: Option<String>,
298    pub ignore_phased_updates: bool,
299    pub aur_command: Option<String>,
300}

302#[derive(Deserialize, Debug, Clone, Copy, PartialEq)]
303#[serde(rename_all = "lowercase")]
304pub enum PackageManager {
305    Apk,
306    Apt,
307    Aur,
308    Brew,
309    Dnf,
310    Pacman,
311    Snap,
312    Xbps,
313}
314
315impl PackageManager {
316    /// The name of the package manager, as used in format strings.
317    fn name(&self) -> &'static str {
318        match self {
319            PackageManager::Apk => "apk",
320            PackageManager::Apt => "apt",
321            PackageManager::Aur => "aur",
322            PackageManager::Brew => "brew",
323            PackageManager::Dnf => "dnf",
324            PackageManager::Pacman => "pacman",
325            PackageManager::Snap => "snap",
326            PackageManager::Xbps => "xbps",
327        }
328    }
329
330    /// Builds a backend for the package manager.
331    async fn build(&self, config: &Config) -> Result<Box<dyn Backend>> {
332        Ok(match self {
333            PackageManager::Apk => Box::new(Apk::new()),
334            PackageManager::Apt => Box::new(Apt::new(config.ignore_phased_updates).await?),
335            PackageManager::Aur => Box::new(Aur::new(
336                config.aur_command.clone().error("aur_command is not set")?,
337            )),
338            PackageManager::Brew => Box::new(Brew::new()),
339            PackageManager::Dnf => Box::new(Dnf::new()),
340            PackageManager::Pacman => Box::new(Pacman::new().await?),
341            PackageManager::Snap => Box::new(Snap::new()),
342            PackageManager::Xbps => Box::new(Xbps::new()),
343        })
344    }
345}

347pub async fn run(config: &Config, api: &CommonApi) -> Result<()> {
348    let mut config: Config = config.clone();
349
350    let format = config.format.with_default(" $icon $total.eng(w:1) ")?;
351    let format_singular = config
352        .format_singular
353        .with_default(" $icon $total.eng(w:1) ")?;
354    let format_up_to_date = config
355        .format_up_to_date
356        .with_default(" $icon $total.eng(w:1) ")?;
357
358    // Check if the user specified a package manager in any format string, then
359    // add that package manager to the config list.
360    macro_rules! check_manager {
361        ($manager:expr) => {{
362            let name = $manager.name();
363            let in_format = format.contains_key(name)
364                || format_singular.contains_key(name)
365                || format_up_to_date.contains_key(name);
366
367            if !config.package_manager.contains(&$manager) && in_format {
368                config.package_manager.push($manager);
369            }
370        }};
371    }
372
373    check_manager!(PackageManager::Apk);
374    check_manager!(PackageManager::Apt);
375    check_manager!(PackageManager::Aur);
376    check_manager!(PackageManager::Brew);
377    check_manager!(PackageManager::Dnf);
378    check_manager!(PackageManager::Pacman);
379    check_manager!(PackageManager::Snap);
380    check_manager!(PackageManager::Xbps);
381
382    let warning_updates_regex = config
383        .warning_updates_regex
384        .as_deref()
385        .map(Regex::new)
386        .transpose()
387        .error("invalid warning updates regex")?;
388    let critical_updates_regex = config
389        .critical_updates_regex
390        .as_deref()
391        .map(Regex::new)
392        .transpose()
393        .error("invalid critical updates regex")?;
394    let ignore_updates_regex = config
395        .ignore_updates_regex
396        .as_deref()
397        .map(Regex::new)
398        .transpose()
399        .error("invalid ignore updates regex")?;
400
401    let mut package_manager_vec: Vec<Box<dyn Backend>> = Vec::new();
402
403    for &package_manager in config.package_manager.iter() {
404        package_manager_vec.push(package_manager.build(&config).await?);
405    }
406
407    loop {
408        let mut package_manager_map: HashMap<Cow<'static, str>, Value> = HashMap::new();
409
410        let mut critical = false;
411        let mut warning = false;
412        let mut total_count = 0;
413
414        // Iterate over the all package manager listed in Config
415        for package_manager in &package_manager_vec {
416            let mut updates = package_manager.get_updates_list().await?;
417            if let Some(regex) = ignore_updates_regex.clone() {
418                updates.retain(|u| !regex.is_match(u));
419            }
420
421            let updates_count = updates.len();
422
423            package_manager_map.insert(package_manager.name(), Value::number(updates_count));
424            total_count += updates_count;
425
426            warning |= warning_updates_regex
427                .as_ref()
428                .is_some_and(|regex| has_matching_update(&updates, regex));
429            critical |= critical_updates_regex
430                .as_ref()
431                .is_some_and(|regex| has_matching_update(&updates, regex));
432        }
433
434        let mut widget = Widget::new();
435
436        package_manager_map.insert("icon".into(), Value::icon("update"));
437        package_manager_map.insert("total".into(), Value::number(total_count));
438
439        widget.set_format(match total_count {
440            0 => format_up_to_date.clone(),
441            1 => format_singular.clone(),
442            _ => format.clone(),
443        });
444        widget.set_values(package_manager_map);
445
446        widget.state = match total_count {
447            0 => State::Idle,
448            _ => {
449                if critical {
450                    State::Critical
451                } else if warning {
452                    State::Warning
453                } else {
454                    State::Info
455                }
456            }
457        };
458        api.set_widget(widget)?;
459
460        select! {
461            _ = sleep(config.interval.0) => (),
462            _ = api.wait_for_update_request() => (),
463        }
464    }
465}

474pub fn has_matching_update(updates: &[String], regex: &Regex) -> bool {
475    updates.iter().any(|line| regex.is_match(line))
