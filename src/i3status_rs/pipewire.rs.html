52pub(crate) enum Direction {
53    Input,
54    Output,
55}
56
57impl FromStr for Direction {
58    type Err = Error;
59
60    fn from_str(s: &str) -> Result<Self, Self::Err> {
61        if s.ends_with("/Source") {
62            Ok(Self::Input)
63        } else if s.ends_with("/Sink") {
64            Ok(Self::Output)
65        } else {
66            Err(Error::new("Invalid media class to determine direction"))
67        }
68    }
69}
70
71impl TryFrom<u32> for Direction {
72    type Error = Error;
73
74    fn try_from(value: u32) -> Result<Self, Self::Error> {
75        match value {
76            SPA_DIRECTION_INPUT => Ok(Self::Input),
77            SPA_DIRECTION_OUTPUT => Ok(Self::Output),
78            _ => Err(Error::new("Invalid direction value, must be 0 or 1")),
79        }
80    }
81}

209struct Proxies<T: ProxyT + 'static> {
210    proxies_t: HashMap<u32, T>,
211    listeners: HashMap<u32, Vec<Box<dyn Listener>>>,
212}
213
214impl<T: ProxyT + 'static> Proxies<T> {
215    fn new() -> Self {
216        Self {
217            proxies_t: HashMap::new(),
218            listeners: HashMap::new(),
219        }
220    }
221
222    fn add_proxy(
223        &mut self,
224        proxy: T,
225        listener: impl Listener + 'static,
226        proxies: &Rc<RefCell<Self>>,
227    ) -> u32 {
228        let listener_spe = Box::new(listener);
229
230        let proxy_upcast = proxy.upcast_ref();
231        let proxy_id = proxy_upcast.id();
232
233        let proxies_weak = Rc::downgrade(proxies);
234
235        let listener = proxy_upcast
236            .add_listener_local()
237            .removed(move || {
238                if let Some(proxies) = proxies_weak.upgrade() {
239                    proxies.borrow_mut().remove(proxy_id);
240                }
241            })
242            .register();
243
244        self.add_proxy_t(proxy_id, proxy, listener_spe);
245        self.add_proxy_listener(proxy_id, listener);
246
247        proxy_id
248    }
249
250    fn add_proxy_t(&mut self, proxy_id: u32, device_proxy: T, listener: Box<dyn Listener>) {
251        self.proxies_t.insert(proxy_id, device_proxy);
252        self.listeners.entry(proxy_id).or_default().push(listener);
253    }
254
255    fn add_proxy_listener(&mut self, proxy_id: u32, listener: ProxyListener) {
256        self.listeners
257            .entry(proxy_id)
258            .or_default()
259            .push(Box::new(listener));
260    }
261
262    fn remove(&mut self, proxy_id: u32) {
263        self.proxies_t.remove(&proxy_id);
264        self.listeners.remove(&proxy_id);
265    }
266}

286#[derive(Clone, Debug)]
287pub(crate) enum CommandKind {
288    Mute(u32, bool),
289    SetVolume(u32, Vec<f32>),
290}
291
292impl CommandKind {
293    fn create_property_value(&self, param_type: ParamType) -> Value {
294        match self {
295            CommandKind::SetVolume(_, volume) => Value::Object(Object {
296                type_: SpaTypes::ObjectParamProps.as_raw(),
297                id: param_type.as_raw(),
298                properties: vec![Property::new(
299                    SPA_PROP_channelVolumes,
300                    Value::ValueArray(ValueArray::Float(
301                        volume
302                            .iter()
303                            .map(|vol| {
304                                let vol = vol / NORMAL;
305                                vol * vol * vol
306                            })
307                            .collect(),
308                    )),
309                )],
310            }),
311            CommandKind::Mute(_, mute) => Value::Object(Object {
312                type_: SpaTypes::ObjectParamProps.as_raw(),
313                id: param_type.as_raw(),
314                properties: vec![Property::new(SPA_PROP_mute, Value::Bool(*mute))],
315            }),
316        }
317    }
318
319    fn execute(
320        self,
321        client: &Client,
322        node_proxies: Rc<RefCell<Proxies<NodeProxy>>>,
323        device_proxies: Rc<RefCell<Proxies<DeviceProxy>>>,
324    ) {
325        debug!("Executing command: {:?}", self);
326        use CommandKind::*;
327        let id = match self {
328            SetVolume(id, _) | Mute(id, _) => id,
329        };
330        let client_data = client.data.lock().unwrap();
331        if let Some(node) = client_data.nodes.get(&id) {
332            if let Some(node_proxy) = node_proxies.borrow_mut().proxies_t.get(&node.proxy_id) {
333                let node_param = self.create_property_value(ParamType::Props);
334                debug!("Setting Node Props param: {:?}", node_param);
335
336                let pod_data = PodSerializer::serialize(Cursor::new(Vec::new()), &node_param)
337                    .expect("Failed to serialize node props pod")
338                    .0
339                    .into_inner();
340                let pod = Pod::from_bytes(&pod_data).expect("Unable to construct pod");
341                node_proxy.set_param(ParamType::Props, 0, pod);
342            }
343
344            if let Some(device_id) = node.device_id
345                && let Some(direction) = node.direction
346                && let Some(directed_routes) = client_data.directed_routes.get(&device_id)
347                && let Some(route) = directed_routes.get_route(direction)
348                && let Some(device_proxy) = device_proxies
349                    .borrow_mut()
350                    .proxies_t
351                    .get(&directed_routes.proxy_id)
352            {
353                let route_param = Value::Object(Object {
354                    type_: SpaTypes::ObjectParamRoute.as_raw(),
355                    id: ParamType::Route.as_raw(),
356                    properties: vec![
357                        Property::new(SPA_PARAM_ROUTE_index, Value::Int(route.index)),
358                        Property::new(SPA_PARAM_ROUTE_device, Value::Int(route.device)),
359                        Property::new(
360                            SPA_PARAM_ROUTE_props,
361                            self.create_property_value(ParamType::Route),
362                        ),
363                        Property::new(SPA_PARAM_ROUTE_save, Value::Bool(true)),
364                    ],
365                });
366                debug!("Setting Device Route param: {:?}", route_param);
367
368                let pod_data = PodSerializer::serialize(Cursor::new(Vec::new()), &route_param)
369                    .expect("Failed to serialize route pod")
370                    .0
371                    .into_inner();
372                let pod = Pod::from_bytes(&pod_data).expect("Unable to construct pod");
373                device_proxy.set_param(ParamType::Route, 0, pod);
374            }
375        }
376    }
377}

379pub(crate) struct Client {
380    event_senders: Mutex<Vec<UnboundedSender<EventKind>>>,
381    command_sender: PwSender<CommandKind>,
382    pub data: Mutex<Data>,
383}
384
385impl Client {
386    fn new() -> Result<Client> {
387        let (tx, rx) = pw_channel();
388
389        thread::Builder::new()
390            .name("i3status_pipewire".to_string())
391            .spawn(|| Client::main_loop_thread(rx))
392            .error("failed to spawn a thread")?;
393
394        Ok(Self {
395            event_senders: Mutex::new(Vec::new()),
396            command_sender: tx,
397            data: Mutex::new(Data::default()),
398        })
399    }
400
401    fn main_loop_thread(command_receiver: PwReceiver<CommandKind>) {
402        let client = CLIENT.as_ref().error("Could not get client").unwrap();
403
404        let proplist = properties! {*keys::APP_NAME => env!("CARGO_PKG_NAME")};
405
406        let main_loop = MainLoopRc::new(None).expect("Failed to create main loop");
407
408        let context = ContextRc::new(&main_loop, Some(proplist)).expect("Failed to create context");
409        let core = context.connect_rc(None).expect("Failed to connect");
410        let registry = core.get_registry_rc().expect("Failed to get registry");
411        let registry_weak = registry.downgrade();
412
413        let update = Rc::new(RefCell::new(EventKind::empty()));
414        let update_copy = update.clone();
415        let update_copy2 = update.clone();
416
417        // Proxies and their listeners need to stay alive so store them here
418        let node_proxies = Rc::new(RefCell::new(Proxies::<NodeProxy>::new()));
419        let node_proxies_weak = Rc::downgrade(&node_proxies);
420        let device_proxies = Rc::new(RefCell::new(Proxies::<DeviceProxy>::new()));
421        let device_proxies_weak = Rc::downgrade(&device_proxies);
422        let metadata_proxies = Rc::new(RefCell::new(Proxies::<MetadataProxy>::new()));
423
424        let _receiver = command_receiver.attach(main_loop.loop_(), move |command: CommandKind| {
425            if let Some(node_proxies) = node_proxies_weak.upgrade()
426                && let Some(device_proxies) = device_proxies_weak.upgrade()
427            {
428                command.execute(client, node_proxies.clone(), device_proxies.clone());
429            }
430        });
431
432        // Register a callback to the `global` event on the registry, which notifies of any new global objects
433        // appearing on the remote.
434        // The callback will only get called as long as we keep the returned listener alive.
435        let _registry_listener = registry
436            .add_listener_local()
437            .global(move |global| {
438                let Some(registry) = registry_weak.upgrade() else {
439                    return;
440                };
441                let global_id = global.id;
442                let Some(global_props) = global.props else {
443                    return;
444                };
445                match &global.type_ {
446                    ObjectType::Node => {
447                        let node_proxy: NodeProxy =
448                            registry.bind(global).expect("Could not bind node");
449                        node_proxy.subscribe_params(&[ParamType::Props]);
450                        let update_copy2 = update_copy.clone();
451                        let update_copy3 = update_copy.clone();
452                        let node_listener = node_proxy
453                            .add_listener_local()
454                            .info(move |info| {
455                                let running = matches!(info.state(), NodeState::Running);
456                                client
457                                    .data
458                                    .lock()
459                                    .unwrap()
460                                    .nodes
461                                    .entry(global_id)
462                                    .and_modify(|node| {
463                                        if node.running != running {
464                                            node.running = running;
465                                            update_copy2.replace_with(|v| {
466                                                *v | EventKind::NODE_STATE_UPDATE
467                                            });
468                                        }
469                                    });
470                            })
471                            .param(move |_seq, _id, _index, _next, param| {
472                                let Some(param) = param else {
473                                    return;
474                                };
475                                let Ok((_, Value::Object(object))) =
476                                    PodDeserializer::deserialize_from::<Value>(param.as_bytes())
477                                else {
478                                    return;
479                                };
480                                client
481                                    .data
482                                    .lock()
483                                    .unwrap()
484                                    .nodes
485                                    .entry(global_id)
486                                    .and_modify(|node| {
487                                        for property in object.properties {
488                                            if property.key == SPA_PROP_mute {
489                                                let Value::Bool(muted) = property.value else {
490                                                    return;
491                                                };
492                                                let muted = Some(muted);
493                                                if node.muted != muted {
494                                                    node.muted = muted;
495                                                    update_copy3.replace_with(|v| {
496                                                        *v | EventKind::NODE_PARAM_UPDATE
497                                                    });
498                                                }
499                                            } else if property.key == SPA_PROP_channelVolumes {
500                                                let Value::ValueArray(ValueArray::Float(volumes)) =
501                                                    property.value
502                                                else {
503                                                    return;
504                                                };
505
506                                                let volume = Some(
507                                                    volumes
508                                                        .iter()
509                                                        .map(|vol| vol.cbrt() * NORMAL)
510                                                        .collect(),
511                                                );
512                                                if node.volume != volume {
513                                                    node.volume = volume;
514                                                    update_copy3.replace_with(|v| {
515                                                        *v | EventKind::NODE_PARAM_UPDATE
516                                                    });
517                                                }
518                                            }
519                                        }
520                                    });
521                            })
522                            .register();
523
524                        let proxy_id = node_proxies.borrow_mut().add_proxy(
525                            node_proxy,
526                            node_listener,
527                            &node_proxies,
528                        );
529
530                        client
531                            .data
532                            .lock()
533                            .unwrap()
534                            .nodes
535                            .insert(global_id, Node::new(global_id, global_props, proxy_id));
536                        update_copy.replace_with(|v| *v | EventKind::NODE_ADDED);
537                    }
538                    ObjectType::Link => {
539                        let mut client_data = client.data.lock().unwrap();
540                        let Some(link) = Link::new(global_props) else {
541                            return;
542                        };
543                        if let Some(node) = client_data.nodes.get(&link.link_input_node)
544                            && node.name == env!("CARGO_PKG_NAME")
545                        {
546                            return;
547                        }
548
549                        client_data.links.insert(global_id, link);
550                        update_copy.replace_with(|v| *v | EventKind::LINK_ADDED);
551                    }
552                    ObjectType::Port => {
553                        client.data.lock().unwrap().ports.insert(global_id);
554                        update_copy.replace_with(|v| *v | EventKind::PORT_ADDED);
555                    }
556                    ObjectType::Device => {
557                        let device_proxy: DeviceProxy =
558                            registry.bind(global).expect("Could not bind device");
559                        device_proxy.subscribe_params(&[ParamType::Route]);
560                        let update_copy2 = update_copy.clone();
561                        let device_listener = device_proxy
562                            .add_listener_local()
563                            .param(move |_seq, _id, _index, _next, param| {
564                                let Some(param) = param else {
565                                    return;
566                                };
567                                let Ok((_, Value::Object(object))) =
568                                    PodDeserializer::deserialize_from::<Value>(param.as_bytes())
569                                else {
570                                    return;
571                                };
572                                let mut route_index = None;
573                                let mut route_direction = None;
574                                let mut route_device = None;
575                                let mut route_name = None;
576                                for property in &object.properties {
577                                    if property.key == SPA_PARAM_ROUTE_index {
578                                        let Value::Int(route_index_v) = property.value else {
579                                            return;
580                                        };
581                                        route_index = Some(route_index_v);
582                                    } else if property.key == SPA_PARAM_ROUTE_direction {
583                                        let Value::Id(route_direction_v) = property.value else {
584                                            return;
585                                        };
586                                        route_direction = route_direction_v.0.try_into().ok();
587                                    } else if property.key == SPA_PARAM_ROUTE_device {
588                                        let Value::Int(route_device_v) = property.value else {
589                                            return;
590                                        };
591                                        route_device = Some(route_device_v);
592                                    } else if property.key == SPA_PARAM_ROUTE_name {
593                                        let Value::String(route_name_v) = property.value.to_owned()
594                                        else {
595                                            return;
596                                        };
597                                        route_name = Some(route_name_v);
598                                    }
599                                }
600
601                                if let Some(route_index) = route_index
602                                    && let Some(route_direction) = route_direction
603                                    && let Some(route_device) = route_device
604                                    && let Some(route_name) = route_name
605                                {
606                                    client
607                                        .data
608                                        .lock()
609                                        .unwrap()
610                                        .directed_routes
611                                        .entry(global_id)
612                                        .and_modify(|directed_routes| {
613                                            let route =
614                                                directed_routes.get_mut_route(route_direction);
615                                            if let Some(route) = route {
616                                                if route.index != route_index
617                                                    || route.device != route_device
618                                                    || route.name != route_name
619                                                {
620                                                    route.index = route_index;
621                                                    route.device = route_device;
622                                                    route.name = route_name;
623                                                    update_copy2.replace_with(|v| {
624                                                        *v | EventKind::DEVICE_PARAM_UPDATE
625                                                    });
626                                                }
627                                            } else {
628                                                route.replace(Route {
629                                                    index: route_index,
630                                                    device: route_device,
631                                                    name: route_name,
632                                                });
633                                                update_copy2.replace_with(|v| {
634                                                    *v | EventKind::DEVICE_PARAM_UPDATE
635                                                });
636                                            }
637                                        });
638                                }
639                            })
640                            .register();
641
642                        let proxy_id = device_proxies.borrow_mut().add_proxy(
643                            device_proxy,
644                            device_listener,
645                            &device_proxies,
646                        );
647
648                        client
649                            .data
650                            .lock()
651                            .unwrap()
652                            .directed_routes
653                            .insert(global_id, DirectedRoutes::new(proxy_id));
654
655                        update_copy.replace_with(|v| *v | EventKind::DEVICE_ADDED);
656                    }
657                    ObjectType::Metadata => {
658                        // There are many kinds of metadata, but we are only interested in the default metadata
659                        if global_props.get("metadata.name") != Some("default") {
660                            return;
661                        }
662                        let metadata_proxy: MetadataProxy =
663                            registry.bind(global).expect("Could not bind device");
664                        let update_copy2 = update_copy.clone();
665                        let metadata_listener = metadata_proxy
666                            .add_listener_local()
667                            .property(move |_subject, key, type_, value| {
668                                if type_ != Some("Spa:String:JSON")
669                                    || (key != Some(DEFAULT_SINK_KEY)
670                                        && key != Some(DEFAULT_SOURCE_KEY))
671                                {
672                                    return -1;
673                                }
674
675                                let Some(value) = value else {
676                                    return -1;
677                                };
678
679                                let value: serde_json::Value =
680                                    serde_json::from_str(value).unwrap_or_default();
681                                let name = value
682                                    .get("name")
683                                    .and_then(|v| v.as_str())
684                                    .map(|s| s.to_string());
685                                if key == Some(DEFAULT_SINK_KEY) {
686                                    let default_medata_sink =
687                                        &mut client.data.lock().unwrap().default_metadata.sink;
688                                    if *default_medata_sink != name {
689                                        *default_medata_sink = name;
690                                        update_copy2.replace_with(|v| {
691                                            *v | EventKind::DEFAULT_META_DATA_UPDATED
692                                        });
693                                    }
694                                } else {
695                                    let default_medata_source =
696                                        &mut client.data.lock().unwrap().default_metadata.source;
697                                    if *default_medata_source != name {
698                                        *default_medata_source = name;
699                                        update_copy2.replace_with(|v| {
700                                            *v | EventKind::DEFAULT_META_DATA_UPDATED
701                                        });
702                                    }
703                                }
704
705                                0
706                            })
707                            .register();
708
709                        metadata_proxies.borrow_mut().add_proxy(
710                            metadata_proxy,
711                            metadata_listener,
712                            &metadata_proxies,
713                        );
714                    }
715                    _ => (),
716                }
717            })
718            .global_remove(move |uid| {
719                let mut client_data = client.data.lock().unwrap();
720                if client_data.nodes.remove(&uid).is_some() {
721                    update_copy2.replace_with(|v| *v | EventKind::NODE_REMOVED);
722                } else if client_data.links.remove(&uid).is_some() {
723                    update_copy2.replace_with(|v| *v | EventKind::LINK_REMOVED);
724                } else if client_data.ports.remove(&uid) {
725                    update_copy2.replace_with(|v| *v | EventKind::PORT_REMOVED);
726                } else if client_data.directed_routes.remove(&uid).is_some() {
727                    update_copy2.replace_with(|v| *v | EventKind::DEVICE_REMOVED);
728                }
729            })
730            .register();
731
732        /* START WORKAROUND FOR PIPEWIRE LOOPBACK BUG
733
734        This workaround fixes an issue caused by the loopback device.
735        Without this workaround, if nothing has interacted with the sink/source then setting
736        the volume/mute will fail.
737
738        This workaround creates input and output streams which forces pipewire
739        to use the real sink/source instead of the loopback device.
740
741        For context see:
742
743            * https://gitlab.freedesktop.org/pipewire/pipewire/-/issues/4949
744            * https://gitlab.freedesktop.org/pipewire/pipewire/-/issues/4610
745            * https://gitlab.freedesktop.org/pipewire/wireplumber/-/issues/822
746        */
747
748        let output_done = Rc::new(Cell::new(false));
749        let output_done_clone = output_done.clone();
750        let input_done = Rc::new(Cell::new(false));
751        let input_done_clone = input_done.clone();
752
753        let values: Vec<u8> = PodSerializer::serialize(
754            Cursor::new(Vec::new()),
755            &Value::Object(Object {
756                type_: SPA_TYPE_OBJECT_Format,
757                id: SPA_PARAM_EnumFormat,
758                properties: AudioInfoRaw::new().into(),
759            }),
760        )
761        .expect("Failed to serialize pod")
762        .0
763        .into_inner();
764
765        let mut params = [Pod::from_bytes(&values).expect("Failed to create pod")];
766
767        let output_stream = StreamBox::new(
768            &core,
769            "i3status_pipewire_workaround_output_stream",
770            properties! {
771                *keys::MEDIA_TYPE => "Audio",
772                *keys::MEDIA_ROLE => "Music",
773                *keys::MEDIA_CATEGORY => "Playback",
774                *keys::AUDIO_CHANNELS => "2",
775            },
776        )
777        .expect("Could not create output_stream");
778
779        let output_stream_listener = output_stream
780            .add_local_listener()
781            .process(move |_stream, _acc: &mut f64| {
782                output_done_clone.set(true);
783            })
784            .register()
785            .expect("Could not add output_stream listener");
786
787        output_stream
788            .connect(
789                ::pipewire::spa::utils::Direction::Output,
790                None,
791                StreamFlags::AUTOCONNECT | StreamFlags::MAP_BUFFERS | StreamFlags::RT_PROCESS,
792                &mut params,
793            )
794            .expect("Could not connect output_stream");
795
796        let input_stream = StreamBox::new(
797            &core,
798            "i3status_pipewire_workaround_input_stream",
799            properties! {
800                *keys::MEDIA_TYPE => "Audio",
801                *keys::MEDIA_ROLE => "Music",
802                *keys::MEDIA_CATEGORY => "Playback",
803                *keys::AUDIO_CHANNELS => "2",
804            },
805        )
806        .expect("Could not create input_stream");
807        let input_stream_listener = input_stream
808            .add_local_listener()
809            .process(move |_stream, _acc: &mut f64| {
810                input_done_clone.set(true);
811            })
812            .register()
813            .expect("Could not add input_stream listener");
814
815        input_stream
816            .connect(
817                ::pipewire::spa::utils::Direction::Input,
818                None,
819                StreamFlags::AUTOCONNECT | StreamFlags::MAP_BUFFERS | StreamFlags::RT_PROCESS,
820                &mut params,
821            )
822            .expect("Could not connect input_stream");
823
824        while !output_done.get() {
825            main_loop.loop_().iterate(Duration::from_secs(1));
826            let event = update.take();
827            if !event.is_empty() {
828                client.send_update_event(event);
829            }
830        }
831        output_stream
832            .disconnect()
833            .expect("Unable to disconnect output_stream");
834        output_stream_listener.unregister();
835
836        while !input_done.get() {
837            main_loop.loop_().iterate(Duration::from_secs(1));
838            let event = update.take();
839            if !event.is_empty() {
840                client.send_update_event(event);
841            }
842        }
843        input_stream
844            .disconnect()
845            .expect("Unable to disconnect input_stream");
846        input_stream_listener.unregister();
847
848        // END WORKAROUND FOR PIPEWIRE LOOPBACK BUG
849
850        loop {
851            main_loop.loop_().iterate(Duration::from_hours(24));
852            let event = update.take();
853            if !event.is_empty() {
854                client.send_update_event(event);
855            }
856        }
857    }
858
859    pub fn add_event_listener(&self, tx: UnboundedSender<EventKind>) {
860        self.event_senders.lock().unwrap().push(tx);
861    }
862
863    pub fn add_command_listener(&self) -> PwSender<CommandKind> {
864        self.command_sender.clone()
865    }
866
867    pub fn send_update_event(&self, event: EventKind) {
868        debug!("send_update_event {:?}", event);
869        self.event_senders
870            .lock()
871            .unwrap()
872            .retain(|tx| tx.send(event).is_ok());
873    }
